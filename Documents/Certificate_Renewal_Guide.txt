
1. Check certificate expiry
echo | openssl s_client -servername api.jackdesk.com -connect api.jackdesk.com:443 2>/dev/null | openssl x509 -noout -dates

2. Stop nginx
sudo systemctl restart nginx

3. Install new certificate for renew
sudo certbot certonly -d api.jackdesk.com -d webhook.jackdesk.com -d websocket.jackdesk.com

4. Restart nginx
sudo systemctl restart nginx
